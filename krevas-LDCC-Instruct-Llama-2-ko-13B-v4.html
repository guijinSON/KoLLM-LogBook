def remove_duplicates(list):
    """
    list: 리스트를 제공하는 목록
    """
    sorted_list = sorted(list)
    seen = set()
    result = []
    for item in sorted_list:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result
